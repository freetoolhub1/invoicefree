━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 1 — CREATE GITHUB ACCOUNT (if not have)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Go to https://github.com
2. Click "Sign up"
3. Enter email, password, username
4. Verify email

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 2 — CREATE GITHUB REPOSITORY
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Log in to GitHub
2. Click the "+" button (top right)
3. Click "New repository"
4. Name it: invoicefree (or any name)
5. Set to PUBLIC
6. DO NOT check "Add README"
7. Click "Create repository"
8. GitHub shows you a page — LEAVE IT OPEN

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 3 — UPLOAD YOUR FILES TO GITHUB
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
EASY METHOD (no coding needed):
1. On the empty repo page, click "uploading an existing file"
   (small link in the middle of the page)
2. Drag ALL files from your invoicefree-v2 folder into the page
   (select all files — NOT the folder itself, the FILES inside it)
   Files to upload:
   - index.html
   - invoice-app.html
   - login.html
   - register.html
   - dashboard.html
   - admin.html
   - upgrade.html
   - pricing.html
   - invoice-view.html
   - privacy.html
   - terms.html
   - sitemap.xml
   - robots.txt
   - vercel.json
   - .gitignore
3. Scroll down, click "Commit changes" (green button)
4. Wait 10 seconds — files are now on GitHub!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 4 — CONNECT VERCEL TO GITHUB
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Go to https://vercel.com
2. Click "Sign Up" → choose "Continue with GitHub"
   (this links your GitHub to Vercel — very important!)
3. Authorize Vercel to access your GitHub
4. You are now inside Vercel dashboard

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 5 — DEPLOY YOUR SITE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. In Vercel, click "Add New Project" (or "New Project")
2. You see a list of your GitHub repos
3. Find "invoicefree" and click "Import"
4. On the next screen:
   - Framework Preset: OTHER (not Next.js or anything else)
   - Root Directory: ./ (leave as default)
   - Build Command: LEAVE EMPTY
   - Output Directory: LEAVE EMPTY
5. Click "Deploy"
6. Wait 30-60 seconds...
7. SUCCESS! Vercel gives you a URL like:
   https://invoicefree-abc123.vercel.app

VISIT THAT URL — your website is live!

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
STEP 6 — CONNECT invoicefree.info DOMAIN
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. Inside your Vercel project, click "Settings"
2. Click "Domains" in the left sidebar
3. Type: invoicefree.info → click "Add"
4. Also add: www.invoicefree.info
5. Vercel shows you DNS records to add. Two options:

   OPTION A — Change nameservers (easier):
   Vercel gives you nameservers like:
     ns1.vercel-dns.com
     ns2.vercel-dns.com
   Go to your domain registrar (GoDaddy/Namecheap/etc)
   Find "Nameservers" for invoicefree.info
   Replace them with Vercel's nameservers

   OPTION B — Add DNS records (if you want to keep registrar DNS):
   Vercel gives you an A record and CNAME:
     A record:     @    →  76.76.21.21
     CNAME record: www  →  cname.vercel-dns.com
   Add these in your domain registrar's DNS settings

6. Wait 24-48 hours for DNS to update worldwide
7. Vercel auto-adds FREE SSL certificate (https://)

STEP 7 — UPDATE YOUR SITE (FUTURE CHANGES)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
Every time you update a file:
1. Go to your GitHub repo
2. Click the file you want to edit
3. Click the pencil icon (Edit)
4. Make changes
5. Click "Commit changes"
6. Vercel AUTOMATICALLY detects the change
7. Auto-deploys in 30 seconds — live instantly!

OR upload new files same way as Step 3.
